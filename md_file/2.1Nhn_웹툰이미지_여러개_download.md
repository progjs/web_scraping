
### 특정 웹툰 페이지의 모든 image를 다운로드
* soup.select('img[src$=.jpg]')
* img 폴더를 생성하고 img 폴더 하위에 파일 저장


```python
import requests
from bs4 import BeautifulSoup
import os

main_url = 'https://comic.naver.com/webtoon/detail.nhn?titleId=746858&no=3&weekday=sun'
res = requests.get(main_url)
html = res.text

soup = BeautifulSoup(html, 'html.parser')
# soup

img_urls = []
for img_url in soup.select("img[src$='.jpg']"):
    print(img_url)
    img_urls.append(img_url['src'])
    
# print(len(img_urls), img_urls)

# 디렉토리 생성
if not os.path.isdir('img'):
    os.mkdir('img')
    
req_header = {
    'referer': main_url
}
for idx, img_url in enumerate(img_urls):
    res2 = requests.get(img_url, headers=req_header)
    img_data = res2.content
    
    file_name = 'image'+str(idx+1)+'.jpg'
    
    with open('img/'+file_name, 'wb') as file:
        print('Writing to {} ({} bytes)'.format(file_name, len(img_data)))
        file.write(img_data)

```
